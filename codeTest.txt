#version 1.0

#define POSITION layout(0)
#define COLOR layout(1)

struct VertexInput {
    var pos: vec3<f32>;
    var col: vec3<f32>;
};

struct VertexOutput {
    var pos: vec3<f32>;
    var col: vec3<f32>;
};

fn vs_main(VertexInput input) -> VertexOutput {
    var output: VertexOutput;

    output.pos = input.pos;
    output.col = input.col;

    for (var i: int = 0; i < 10; i++) {
        println("Hello, World! x {}", i);
    }

    return output;
}
